\begin{block}{EU-BRIDGE}
   \begin{itemize}
   \item Research project funded by the European Union \url{http://www.eu-bridge.eu}
    \vspace*{0.5ex}
   \item EU-BRIDGE IWSLT 2014 single systems:
    \vspace*{0.5ex}
    \begin{description}
    \item[RWTH:] \hspace*{0.5ex} RWTH Aachen University
    \item[UEDIN:] \hspace*{0.5ex} University of Edinburgh
    \item[KIT:] \hspace*{0.5ex} Karlsruhe Institute of Technology
    \item[FBK:] \hspace*{0.5ex} Fondazione Bruno Kessler
    \end{description}
    \vspace*{0.5ex}
    \item Combined EU-BRIDGE IWSLT 2014 submissions \\via the RWTH system combination approach
    \item Our system combination framework is publicly available as art of release 2.3 of the \textbf{Jane} toolkit \url{http://www.hltpr.rwth-aachen.de/jane/}
    \end{itemize}
    \vspace{-1.0ex}

\end{block}

\begin{block}{RWTH Aachen University}

  \begin{itemize}
    \item Phrase-based systems based on open source Jane toolkit
    \item Trained on all available data
    \item Three backoff-language models: 5-gram general domain, 5-gram in-domain, 7-gram word class LM
    \item Hierarchical reordering model
    \item Maximum expected \BLEU training (phrase, lexical and reordering features)
    \item Recurrent neural language and bidirectional translation models
    \item SLT: Punctuation prediction and recasing with monolingual hiero system
  \end{itemize}
    \vspace{-1.0ex}
\end{block}




\begin{block}{Karlsruhe Institute of Technology}
      \begin{itemize}
              \item Phrase-based translation systems trained on large corpora
              \item Systems for English$\to$German and English$\to$French MT, as well as German$\to$English MT and SLT
              \item Data selection and filtering using cross entropy for \\ language modeling
              \item Combination of language models (LM): word-based LMs, POS-based LM,  and cluster LM
              \item Combination of reordering models: lexicalized reordering, POS-based and tree-based reordering rules
              \item Source and target side contexts are modeled with bilingual LMs and discriminative word lexica using source context
              \item Punctuation insertion for ASR output with a monolingual translation system
      \end{itemize}
       \vspace{-1.0ex}
\end{block}

\begin{block}{University of Edinburgh}

\begin{itemize}
\item Moses with advanced features
\vspace*{0.5ex}
\item SLT system:
  \vspace*{0.5ex}
  \begin{itemize}
    \item Monolingual, monotone SMT punctuation insertion model for punctuation of ASR output 
    \item Phrase-based translation system with operation sequence model over words and word clusters, sparse lexical features and hierarchical lexicalized reordering
  \end{itemize}
\vspace*{0.5ex}
\item MT systems:
  \vspace*{0.5ex}
  \begin{itemize}
    \item Phrase-based with factored translation model\\ (DeEn, EnFr UEDIN-A~\&~B, EnDe UEDIN-A~\&~B)
    \item String-to-tree syntax-based (EnDe UEDIN-C)
    \item EnFr: UEDIN-B trained with all corpora, UEDIN-A without Multi\-UN and $10^9$
    \item EnDe: UEDIN-B trained with parallel data restricted to the in-domain WIT$^3$ corpus, UEDIN-A~\&~C trained with all corpora
  \end{itemize}

\end{itemize}

          \item Selected data: 57M English, 63M French words
